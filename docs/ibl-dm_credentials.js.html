import HttpService from "../http";
import endpoint from "../config";

const http = new HttpService();

/**
 * Retrieves the credentials for an organisation
 * @kind function
 * @name getOrgCredentials
 *
 * @param {Object} data
 */
export function getOrgCredentials(data, successCb = null) {
  http.get(
    endpoint.IBL_DM.GET_ORG_CREDENTIALS,
    data,
    (data) => {
      successCb(data);
    },
    (error) => {
      console.error("Error occurred:", error);
    }
  );
}

export function getUserCredentials(

}

export function getCredential(data, successCb = null) {
  http.get(
    endpoint.IBL_DM.GET_CREDENTIAL(data.entity_id),
    data,
    (data) => {
      successCb(data);
    },
    (error) => {
      console.error("Error occurred:", error);
    }
  );
}
