/**
 * @namespace iblweblogout
 */

import HttpService from "../http";
import endpoint from "../config";

const http = new HttpService();

/**
 * Logout a user from the entire platform
 * @function logout
 * @memberof iblweblogout
 * @param {String} redirect_url URL the user should be brought back to after logout
 * @example <caption>Logout user</caption>
 * This usage:
 *
 * ```js
 * iblweblogout.logout()
 * ```
 */
export function logout(data, callback = null) {
  http.post(
    endpoint.LOGOUT,
    data,
    (data) => {
      callback && callback(data);
    },
    (error) => {
      console.error("Error occurred:", error);
    },
    true
  );
}
